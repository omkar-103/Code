    Solution to the producer-Consumer problem using Shared Memory.

                    import multiprocessing
                import time
                import random
                
                def producer(queue):
                    for _ in range(5):
                        item = random.randint(1, 100)
                        queue.put(item)  # Produce an item
                        print(f"Produced: {item}")
                        time.sleep(1)
                
                def consumer(queue):
                    for _ in range(5):
                        item = queue.get()  # Consume an item
                        print(f"Consumed: {item}")
                        time.sleep(1)
                
                if __name__ == '__main__':
                    queue = multiprocessing.Queue(maxsize=5)  # Shared queue
                    p = multiprocessing.Process(target=producer, args=(queue,))
                    c = multiprocessing.Process(target=consumer, args=(queue,))
                    
                    p.start()
                    c.start()
                    
                    p.join()
                    c.join()

    Solution to the producer-Consumer problem using message passing. 

                    import threading
                import time
                import queue
                
                # Shared queue
                buffer = queue.Queue(maxsize=5)
                
                def producer():
                    for i in range(10):
                        item = f"Item {i}"
                        buffer.put(item)  # Produce an item
                        print(f"Produced: {item}")
                        time.sleep(1)
                
                def consumer():
                    for _ in range(10):
                        item = buffer.get()  # Consume an item
                        print(f"Consumed: {item}")
                        buffer.task_done()  # Mark the task as done
                        time.sleep(1)
                
                # Create and start threads
                producer_thread = threading.Thread(target=producer)
                consumer_thread = threading.Thread(target=consumer)
                
                producer_thread.start()
                consumer_thread.start()
                
                producer_thread.join()
                consumer_thread.join()

    a solution to the readerswriters problem.

                    import threading
                import time
                
                data = 0
                read_count = 0
                mutex = threading.Lock()
                write_lock = threading.Lock()
                
                def reader(id):
                    global read_count
                    while True:
                        mutex.acquire()
                        read_count += 1
                        if read_count == 1:
                            write_lock.acquire()
                        mutex.release()
                
                        print(f"Reader {id} reads data: {data}")
                        time.sleep(1)
                
                        mutex.acquire()
                        read_count -= 1
                        if read_count == 0:
                            write_lock.release()
                        mutex.release()
                        time.sleep(1)
                
                def writer(id):
                    global data
                    while True:
                        write_lock.acquire()
                        data += 1
                        print(f"Writer {id} writes data: {data}")
                        write_lock.release()
                        time.sleep(2)
                
                # Create and start threads
                threading.Thread(target=reader, args=(1,)).start()
                threading.Thread(target=reader, args=(2,)).start()
                threading.Thread(target=writer, args=(1,)).start()

    FCFS scheduling algorithm. (with no pre-emption) scheduling algorithm. 

            def fcfs_scheduling(processes, burst_times):
        n = len(processes)
        waiting_time = [0] * n
        turnaround_time = [0] * n
    
        for i in range(1, n):
            waiting_time[i] = waiting_time[i - 1] + burst_times[i - 1]
    
        for i in range(n):
            turnaround_time[i] = waiting_time[i] + burst_times[i]
    
        print("Process\tBurst Time\tWaiting Time\tTurnaround Time")
        for i in range(n):
            print(f"{processes[i]}\t{burst_times[i]}\t\t{waiting_time[i]}\t\t{turnaround_time[i]}")
    
    # Example usage
    processes = ['P1', 'P2', 'P3']
    burst_times = [10, 5, 8]
    fcfs_scheduling(processes, burst_times)

    RR Scheduling Algorithm..

            def round_robin(processes, burst_times, time_quantum):
        n = len(processes)  # Number of processes
        remaining_times = burst_times.copy()  # Remaining burst times for each process
        waiting_times = [0] * n  # Waiting times for each process
        turnaround_times = [0] * n  # Turnaround times for each process
        time = 0  # Current time
    
        while True:
            done = True
            for i in range(n):
                if remaining_times[i] > 0:
                    done = False  # There are still processes to execute
    
                    if remaining_times[i] > time_quantum:
                        time += time_quantum
                        remaining_times[i] -= time_quantum
                    else:
                        time += remaining_times[i]
                        waiting_times[i] = time - burst_times[i]
                        remaining_times[i] = 0
    
            if done:  # If all processes are done
                break
    
        for i in range(n):
            turnaround_times[i] = burst_times[i] + waiting_times[i]
    
        return waiting_times, turnaround_times
    
    # Example usage
    processes = ['P1', 'P2', 'P3']
    burst_times = [24, 3, 3]
    time_quantum = 4
    
    waiting_times, turnaround_times = round_robin(processes, burst_times, time_quantum)
    
    print("Process\tBurst Time\tWaiting Time\tTurnaround Time")
    for i in range(len(processes)):
        print(f"{processes[i]}\t{burst_times[i]}\t\t{waiting_times[i]}\t\t{turnaround_times[i]}")

    FIFO page-replacement algorithm. .

           def fifo_page_replacement(pages, frames):
    memory = []
    page_faults = 0
    index = 0

    for page in pages:
        if page not in memory:
            if len(memory) < frames:
                memory.append(page)  # Add page if there's space
            else:
                memory[index] = page  # Replace the oldest page
                index = (index + 1) % frames  # Move the pointer in a circular manner
            page_faults += 1

        print(f"Memory: {memory}")

    return page_faults

# Example usage
pages = [7, 0, 1, 2, 0, 3, 0, 4, 2, 3, 0, 3, 2]
frames = 3
result = fifo_page_replacement(pages, frames)
print(f"Total page faults: {result}")

    LRU page-replacement algorithm. .

            def lru_page_replacement(pages, frames):
        memory = []
        page_faults = 0
    
        for page in pages:
            if page not in memory:
                if len(memory) < frames:
                    memory.append(page)  # add page if there's space
                else:
                    memory.pop(0)  # remove least recently used page
                    memory.append(page)  # add new page
                page_faults += 1
            else:
                # move page to the end to mark it as recently used
                memory.remove(page)
                memory.append(page)
    
            print(f"Memory: {memory}")
    
        return page_faults
    
    # Example usage
    pages = [7, 0, 1, 2, 0, 3, 0, 4, 2, 3, 0, 3, 2]
    frames = 3
    result = lru_page_replacement(pages, frames)
    print(f"Total page faults: {result}")

    Solution to buffer problem

            import threading, time, random

buffer = []
buffer_size = 5
empty = threading.Semaphore(buffer_size)
full = threading.Semaphore(0)
mutex = threading.Lock()

def producer():
    while True:
        item = random.randint(1, 100)
        empty.acquire()
        mutex.acquire()
        buffer.append(item)
        print(f"Produced: {item}")
        mutex.release()
        full.release()
        time.sleep(1)

def consumer():
    while True:
        full.acquire()
        mutex.acquire()
        item = buffer.pop(0)
        print(f"Consumed: {item}")
        mutex.release()
        empty.release()
        time.sleep(1)

threading.Thread(target=producer).start()
threading.Thread(target=consumer).start()
